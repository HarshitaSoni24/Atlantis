STAGE 1: Add Toggle Button

Style: 3 horizontal lines (hamburger icon) or use â˜° symbol

STAGE 2: Modify Sidebar Container
Update your existing sidebar with these properties:

Add a CSS class for collapsed state
Use CSS transforms for smooth sliding animation
Keep the same dark background and styling
Width when expanded: 250px-280px
Width when collapsed: 0px (or -280px left position)

.sidebar {
  position: fixed;
  left: 0;

STAGE 4: JavaScript Toggle Functionality
Add this JavaScript to control the sidebar:

javascript
// Get elements
const sidebarToggle = document.querySelector('.sidebar-toggle');
const sidebar = document.querySelector('.sidebar');
const mainContent = document.querySelector('.main-content');

// Toggle function
function toggleSidebar() {
  sidebar.classList.toggle('collapsed');
  mainContent.classList.toggle('sidebar-collapsed');
  sidebarToggle.classList.toggle('collapsed');
}

// Add click event listener
sidebarToggle.addEventListener('click', toggleSidebar);

// Optional: Close sidebar when clicking outside (for mobile)
document.addEventListener('click', function(event) {
  if (window.innerWidth <= 768) { // Only on mobile
    if (!sidebar.contains(event.target) && !sidebarToggle.contains(event.target)) {
      sidebar.classList.add('collapsed');
      mainContent.classList.add('sidebar-collapsed');
      sidebarToggle.classList.add('collapsed');
    }
  }
});

Make it mobile-friendly with these media queries:

    width: 100%;
    transform: translateX(-100%);
  }

  }
  
  .sidebar:not(.collapsed) {
    transform: translateX(0);
  }
  
  .main-content {
    margin-left: 0;
  }
  
  .sidebar-toggle {
    left: 20px;
  }
  
  .sidebar-toggle.collapsed {
    left: 20px;
  }
}
